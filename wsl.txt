distribution -->

	list --> wsl -l -v

	default --> wsl --set-default <distribution-name>

	start --> wsl -d <distribution-name>
	stop --> wsl --shutdown

	version --> wsl --set-version <distribution-name> 2
	update -->  wsl --update

	delete -->  wsl --unregister <distribution-name>
