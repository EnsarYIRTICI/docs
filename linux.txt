	iso repair -->

		sudo su -
		fdisk -l
		mount /dev/nvme0n1p5 /mnt
		mount /dev/nvme0n1p1 /mnt/boot/efi

		for i in /dev /dev/pts /proc /sys /run; do mount -B $i /mnt$i; done

		chroot /mnt
		mount -t efivarfs none /sys/firmware/efi/efivars
		grub-install /dev/nvme0n1
		update-grub
