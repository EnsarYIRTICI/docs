nginx --> 
	yardımcılar --> apt update
			apt install micro

	yeniden başlat --> nginx -s reload

	genel ayarlar -->

		sudo micro /etc/nginx/nginx.conf

		http {
    			log_format custom '$remote_addr - $remote_user [$time_local] "$request" '
                      			'$status $body_bytes_sent "$http_referer" '
                      			'"$http_user_agent" "$http_x_forwarded_for"';
		}

	server ayarları -->

		sudo micro /etc/nginx/sites-available/default
		micro /etc/nginx/conf.d/default.conf

		server {
    			server_name  ddmvcp.com;

    			location / {
        			proxy_pass http://192.168.1.111:3030;

           			proxy_set_header Host $host;
            			proxy_set_header X-Real-IP $remote_addr;
           			proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
           			proxy_set_header X-Forwarded-Proto $scheme;
    			}

			access_log /var/log/nginx/ddmvcp_access.log custom;

		}

open-ssl -->

	openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
	-keyout /etc/ssl/private/selfsigned.key \
	-out /etc/ssl/certs/selfsigned.crt

	server {
    		listen 80;
    		server_name example.com;
    		return 301 https://$host$request_uri;
	}

	server {
    		listen 443 ssl;
    		server_name example.com;

    		ssl_certificate /etc/ssl/certs/selfsigned.crt;
    		ssl_certificate_key /etc/ssl/private/selfsigned.key;

    		location / {
        		root /var/www/html;
        		index index.html index.htm;
    		}
	}
